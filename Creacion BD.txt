CREATE TABLE USERS
(
ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1000, INCREMENT BY 10),
USERNAME VARCHAR(24) NOT NULL UNIQUE,
PWD VARCHAR(24) NOT NULL,
PRIMARY KEY(ID)
)

CREATE TABLE FOLLOWING
(
FOLLOWED_NAME VARCHAR(24) REFERENCES USERS(USERNAME),
FOLLOWER_NAME VARCHAR(24) REFERENCES USERS(USERNAME),
PRIMARY KEY(FOLLOWED_NAME, FOLLOWER_NAME)
)
